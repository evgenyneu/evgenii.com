\begin{equation*}
  F = \frac{G m_1 m_2}{r^2}
\end{equation*}


\begin{equation*}
  \ddot{\vec{r}} = - \frac{G (m_1 + m_2)}{r^2} \hat{r}
\end{equation*}

\begin{equation*}
  \ddot{\vec{r}} = - \frac{1 + q}{r^2} \hat{r}
\end{equation*}

\begin{align*}
  \ddot{x} &= - \frac{1 + q}{r^3} x \\
  \ddot{y} &= - \frac{1 + q}{r^3} y
\end{align*}


\begin{align*}
  u_1 &= x \\
  u_2 &= y \\
  u_3 &= \dot{x} \\
  u_4 &= \dot{y}
\end{align*}


\begin{align*}
  \dot{u}_1 &= u_3 \\
  \dot{u}_2 &= u_4 \\
  \dot{u}_3 &= - \frac{1 + q}{r^3} u_1 \\
  \dot{u}_4 &= - \frac{1 + q}{r^3} u_2
\end{align*}
